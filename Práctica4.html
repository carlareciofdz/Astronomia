# Apartado A
# we are going to plot the light curve of 3 diferent stars
# the data is in teh file: Fluxes.cvs
# the file has 4 colums: time in Julian days, intensity of a star A, intensity of star B, intensity of star C

#Load the data
data =pd.read_csv("Fluxes.csv",skiprows=3, header=None, names = ["JulianDay", "A", "B","C"])
print(data.columns)

# Apartado 2
# Transform julian time to modified julian time
data["MJD"] = data ["JulianDay"] - 2400000.5
print(data)

# Apartado 3
# Represent the fluxes of stars A, B and C in modified julian time. 
# Use a diferent colour for each star, points not lines and add a leyend, name the exes and a title.

plt.scatter(data["MJD"], data["A"], color='red', marker='o', linestyle='None', label='Star A', s=10)
plt.scatter(data["MJD"], data["B"], color='green', marker='o', linestyle='None', label='Star B',s=10)
plt.scatter(data["MJD"], data["C"], color='blue', marker='o', linestyle='None', label='Star C',s=10)

plt.xlabel('Modified Julian Time')
plt.ylabel('Flux')
plt.title('Fluxes of Stars A, B, and C')
plt.legend()

plt.show()

# Apartado 4
# calculate the mean flux of each star and the standard deviation of the fluxes of each star
mean_A = data["A"].mean()
std_A = data["A"].std()
print("El flujo medio de la estrella A es: ", mean_A, "y la desviaci√≥n estandar es: ", std_A)

mean_B = data["B"].mean()
std_B = data["B"].std()
print("El flujo medio de la estrella B es: ", mean_B, "y la desviaci√≥n estandar es: ", std_B)

mean_C = data["C"].mean()
std_C = data["C"].std()
print("El flujo medio de la estrella C es: ", mean_C, "y la desviaci√≥n estandar es: ", std_C)
# which star has the most precise measurements?
if std_A < std_B and std_A < std_C:
    print("La estrella medida con mayor precisi√≥n es la A")
elif std_B < std_A and std_B < std_C:
    print("La estrella medida con mayor precisi√≥n es la B")
else:
    print("La estrella medida con mayor precisi√≥n es la C")

# Apartado 5
# Calculate the variation coefficient of the fluxes of each star
variation_A = std_A/mean_A
variation_B = std_B/mean_B
variation_C = std_C/mean_C
print("El coeficiente de variaci√≥n de la estrella A es: ", variation_A,"el de la estrella B es: ", variation_B, "y el de la estrella C es: ", variation_C)
# Which star has the most precise measurements?
if variation_A < variation_B and variation_A < variation_C:
    print("La estrella medida con mayor precisi√≥n es la A")
elif variation_B < variation_A and variation_B < variation_C:
    print("La estrella medida con mayor precisi√≥n es la B")
else:
    print("La estrella medida con mayor precisi√≥n es la C")

# Apartado 6
